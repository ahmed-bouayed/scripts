apt-get install cloud-init

wget -q https://cloud-images.ubuntu.com/jammy/current/jammy-server-cloudimg-amd64.img

qm create 8000 --memory 2048 --core 2 --name ubuntu-cloud --net0 virtio,bridge=vmbr0
qm importdisk 8000 jammy-server-cloudimg-amd64.img HDD
qm set 8000 --scsihw virtio-scsi-pci --scsi0 local:vm-9000-disk-0
qm set 8000 --ide2 local-lvm:cloudinit
qm set 8000 --boot c --bootdisk scsi0
qm set 8000 --serial0 socket --vga serial0



---
/var/lib/vz/snippets/vendor.yaml

#cloud-config
runcmd:
    - apt update
    - apt install -y qemu-guest-agent
    - systemctl start qemu-guest-agent
    - reboot
---

qm set 8000 --cicustom "vendor=local:snippets/vendor.yaml"
qm template 8000


sudo rm -f /etc/machine-id
sudo rm -f /var/lib/dbus/machine-id
sudo systemd-machine-id-setup
